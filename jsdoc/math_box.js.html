/**
 * @fileoverview A utility class for representing a numeric box.
 */
'use strict';

class MathBox {
  /**
   * Class for representing a box. A box is specified as a top, right, bottom,
   * and left. A box is useful for representing margins and padding.
   *
   * This class assumes 'screen coordinates': larger Y coordinates are further
   * from the top of the screen.
   *
   * @param {number} top Top.
   * @param {number} right Right.
   * @param {number} bottom Bottom.
   * @param {number} left Left.
   * @struct
   * @constructor
   */
  constructor(top, right, bottom, left) {
    /**
     * Top
     * @type {number}
     */
    this.top = top;

    /**
     * Right
     * @type {number}
     */
    this.right = right;

    /**
     * Bottom
     * @type {number}
     */
    this.bottom = bottom;

    /**
     * Left
     * @type {number}
     */
    this.left = left;
  }

  static boundingBox(...coordinates) {
    /**
     * Creates a Box by bounding a collection of MathCoordinate objects
     * @param {MathCoordinate[]} coordinates - Coordinates to be included inside
     *     the box.
     * @return {!MathBox} - A Box containing all the specified Coordinates.
     */
    const first = coordinates[0];
    let box = new MathBox(first.y, first.x, first.y, first.x);
    for (let i = 1; i < coordinates.length; i++) {
      box.expandToIncludeCoordinate(coordinates[i]);
    }
    return box;
  }

  expandToIncludeCoordinate(coord) {
    /**
     * Expand this box to include the coordinate.
     * @param {!MathCoordinate} coord The coordinate to be included
     *     inside the box.
     */
    this.top = Math.min(this.top, coord.y);
    this.right = Math.max(this.right, coord.x);
    this.bottom = Math.max(this.bottom, coord.y);
    this.left = Math.min(this.left, coord.x);
  }
}
